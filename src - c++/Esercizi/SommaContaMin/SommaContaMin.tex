\section{Somma e Conta Min}
Dotare la classe \cod{AlberoBinario} (\cfr{Ex:Albero Binario}) dei metodi conformi alla seguente specifica.

\begin{methodslist}

\method{Somma}{TElem}{TElem} {
Riceve in ingresso un valore \cod{min} di tipo \cod{TElem} e restituisce la somma degli elementi presenti
nell'albero maggiori o uguali a \cod{min}.
}

\method{NumElem}{TElem}{unsigned int} {
Riceve in ingresso un valore \cod{min} di tipo \cod{TElem} e restituisce il numero degli elementi presenti
nell'albero maggiori o uguali a \cod{min}.
}

\end{methodslist}
